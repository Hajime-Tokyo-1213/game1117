# 改修仕様書 41

## フェーズ2.16 - AuthContextのAPIクライアント統合

### 改修の目的
認証コンテキストを新しいAPIクライアントを使用するように更新

### 改修内容

#### AuthContext.jsxの完全リファクタリング
```javascript
// src/contexts/AuthContext.jsx
import { createContext, useContext, useReducer, useEffect } from 'react';
import { authClient, ApiError } from '../utils/api.js';
import Cookies from 'js-cookie';

const AuthContext = createContext();

// 認証状態管理用reducer
const authReducer = (state, action) => {
  switch (action.type) {
    case 'SET_LOADING':
      return { ...state, loading: action.payload };
    case 'LOGIN_SUCCESS':
      return { ...state, user: action.payload, isAuthenticated: true, loading: false };
    case 'LOGOUT':
      return { ...state, user: null, isAuthenticated: false, loading: false };
    case 'SET_ERROR':
      return { ...state, error: action.payload, loading: false };
    case 'CLEAR_ERROR':
      return { ...state, error: null };
    default:
      return state;
  }
};

export const AuthProvider = ({ children }) => {
  const [state, dispatch] = useReducer(authReducer, {
    user: null,
    isAuthenticated: false,
    loading: true,
    error: null
  });

  // 初期化時のトークン検証
  useEffect(() => {
    const initializeAuth = async () => {
      const token = Cookies.get('authToken');
      
      if (!token) {
        dispatch({ type: 'SET_LOADING', payload: false });
        return;
      }

      try {
        authClient.setAuthToken(token);
        const response = await authClient.verifyToken();
        
        if (response.valid) {
          dispatch({ type: 'LOGIN_SUCCESS', payload: response.user });
        } else {
          Cookies.remove('authToken');
          dispatch({ type: 'LOGOUT' });
        }
      } catch (error) {
        console.error('Token verification failed:', error);
        Cookies.remove('authToken');
        dispatch({ type: 'LOGOUT' });
      }
    };

    initializeAuth();
  }, []);

  const login = async (email, password) => {
    try {
      dispatch({ type: 'SET_LOADING', payload: true });
      dispatch({ type: 'CLEAR_ERROR' });

      const response = await authClient.login(email, password);
      
      // JWTトークンをhttpOnlyクッキーに保存
      Cookies.set('authToken', response.token, {
        httpOnly: false, // クライアントサイドでアクセス可能
        secure: process.env.NODE_ENV === 'production',
        sameSite: 'strict',
        expires: 7 // 7日間
      });

      dispatch({ type: 'LOGIN_SUCCESS', payload: response.user });
      
      return response;
    } catch (error) {
      const errorMessage = error instanceof ApiError 
        ? error.message 
        : '認証に失敗しました';
      
      dispatch({ type: 'SET_ERROR', payload: errorMessage });
      throw error;
    }
  };

  const logout = async () => {
    try {
      await authClient.logout();
    } catch (error) {
      console.error('Logout API call failed:', error);
    } finally {
      Cookies.remove('authToken');
      dispatch({ type: 'LOGOUT' });
    }
  };

  const register = async (userData) => {
    try {
      dispatch({ type: 'SET_LOADING', payload: true });
      dispatch({ type: 'CLEAR_ERROR' });

      const response = await authClient.register(userData);
      return response;
    } catch (error) {
      const errorMessage = error instanceof ApiError 
        ? error.message 
        : '登録に失敗しました';
      
      dispatch({ type: 'SET_ERROR', payload: errorMessage });
      throw error;
    } finally {
      dispatch({ type: 'SET_LOADING', payload: false });
    }
  };

  const updateProfile = async (profileData) => {
    try {
      const response = await authClient.updateProfile(profileData);
      dispatch({ type: 'LOGIN_SUCCESS', payload: response.user });
      return response;
    } catch (error) {
      const errorMessage = error instanceof ApiError 
        ? error.message 
        : 'プロフィール更新に失敗しました';
      
      dispatch({ type: 'SET_ERROR', payload: errorMessage });
      throw error;
    }
  };

  const value = {
    ...state,
    login,
    logout,
    register,
    updateProfile,
    clearError: () => dispatch({ type: 'CLEAR_ERROR' })
  };

  return (
    <AuthContext.Provider value={value}>
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be used within an AuthProvider');
  }
  return context;
};
```
